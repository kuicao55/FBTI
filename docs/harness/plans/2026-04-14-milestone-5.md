# Add Restaurant Recommendation Feature Implementation Plan

> **Harness note:** This plan is executed via `harness-execution` using the Orchestrator / Executor / Reviewer architecture. Each task goes through Executor (TDD implementation) → Spec Reviewer (compliance check) → Code Quality Reviewer (adversarial review). Only Code Quality Review PASS closes a task.

**Goal:** Add restaurant recommendation feature to result page: users can submit restaurant recommendations per personality type (80 types, e.g. HUAE), and other users with the same type see those recommendations.

**Milestone ref:** milestone-5 from claude-progress.json

**Architecture:** Static website with JSON data file. `data/restaurants.json` stores all restaurant recommendations keyed by 4-letter type code. User submits recommendation → triggers JSON file download for manual deploy. `data/restaurant-settings.json` stores configurable limits.

**Tech Stack:** Vanilla JS (ES Modules), HTML5, CSS3, no new dependencies

- [ ] **Step 6: Add submitRestaurant function** (insert near end of script, before `showToast`)

```javascript
async function submitRestaurant(name, nick) {
  var typeCode = window.currentRestaurantType;
  if (!typeCode || !name) return;

  var settings = window.restaurantSettings || { maxPerType: 5 };
  var max = settings.maxPerType || 5;

  // Get current restaurants from localStorage or rest array
  var stored = null;
  try { stored = JSON.parse(localStorage.getItem('fbti_restaurants') || 'null'); } catch(e) {}
  var rest = stored || {};
  if (!Array.isArray(rest[typeCode])) rest[typeCode] = [];

  if (rest[typeCode].length >= max) {
    showToast('该类型已达推荐上限');
    closeRestaurantModal();
    return;
  }

  var today = new Date();
  var dateStr = today.getFullYear() + '-' + String(today.getMonth() + 1).padStart(2, '0') + '-' + String(today.getDate()).padStart(2, '0');

  var newEntry = { name: name, by: nick, date: dateStr };
  rest[typeCode].push(newEntry);

  try { localStorage.setItem('fbti_restaurants', JSON.stringify(rest)); } catch(e) {}

  closeRestaurantModal();
  showToast('推荐已添加 ✓');

  // Re-render restaurant section
  var container = document.getElementById('restaurant-section-container');
  if (container) {
    container.innerHTML = '';
    var section = rest[typeCode].length > 0 ? rest[typeCode] : [];
    if (typeof renderRestaurantSection === 'function' && section.length > 0) {
      renderRestaurantSection(container, typeCode, section);
    }
  }

  // Trigger JSON download for manual deploy
  downloadRestaurantsJson(rest);
}

function downloadRestaurantsJson(rest) {
  var blob = new Blob([JSON.stringify(rest, null, 2)], { type: 'application/json' });
  var url = URL.createObjectURL(blob);
  var a = document.createElement('a');
  a.href = url;
  a.download = 'restaurants.json';
  a.click();
  setTimeout(function() { URL.revokeObjectURL(url); }, 1000);
}
```
