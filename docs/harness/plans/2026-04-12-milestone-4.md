// modules/scoring.js — TasteType 3.1 scoring engine

const DIMENSION_CONFIG = {
  stimulus: { codes: ['H','N','C','M'], minPerTendency: [1,1,1,1] },
  taste: { codes: ['U','S','W','B','O'], minPerTendency: [1,1,1,1,1] },
  philosophy: { codes: ['A','S'], minPerTendency: [1,1] },
  novelty: { codes: ['E','C'], minPerTendency: [2,2] }
};

export function selectQuestions(questionsData) {
  const byDimension = {};
  for (const dim of questionsData.dimensions) {
    byDimension[dim] = questionsData.questions.filter(q => q.dimension === dim);
  }

  const selected = [];
  for (const [dim, config] of Object.entries(DIMENSION_CONFIG)) {
    const pool = [...byDimension[dim]];
    const needed = dim === 'stimulus' ? 6 : dim === 'taste' ? 8 : dim === 'philosophy' ? 4 : 6;
    const minPer = config.minPerTendency;
    const codes = config.codes;

    // Ensure minimum per tendency
    const guaranteed = [];
    for (let i = 0; i < codes.length; i++) {
      const candidates = pool.filter(q =>
        q.options.some(o => o.scores && o.scores[codes[i]] > 0)
      );
      if (candidates.length > 0) {
        const picked = candidates[Math.floor(Math.random() * candidates.length)];
        guaranteed.push(picked);
        pool.splice(pool.indexOf(picked), 1);
      }
    }

    // Fill remaining slots randomly
    while (selected.length < needed + Object.keys(DIMENSION_CONFIG).indexOf(dim) * 0) {

    }

    selected.push(...guaranteed);
  }

  // Shuffle
  return selected.sort(() => Math.random() - 0.5);
}

export function calculateScores(answers, questions) {
  // answers: array of option labels ['A','B',...]
  // questions: selected 24 questions
  const dimensionScores = {
    stimulus: { H:0, N:0, C:0, M:0 },
    taste: { U:0, S:0, W:0, B:0, O:0 },
    philosophy: { A:0, S:0 },
    novelty: { E:0, C:0 }
  };

  questions.forEach((q, i) => {
    const answer = answers[i];
    const option = q.options.find(o => o.label === answer);
    if (option && option.scores) {
      for (const [code, score] of Object.entries(option.scores)) {
        if (dimensionScores[q.dimension] && dimensionScores[q.dimension][code] !== undefined) {
          dimensionScores[q.dimension][code] += score;
        }
      }
    }
  });

  return dimensionScores;
}

export function calculatePercentages(scores) {
  const percentages = {};
  for (const [dim, codeScores] of Object.entries(scores)) {
    const total = Object.values(codeScores).reduce((a, b) => a + b, 0);
    percentages[dim] = {};
    for (const [code, score] of Object.entries(codeScores)) {
      percentages[dim][code] = total > 0 ? (score / total) * 100 : 0;
    }
  }
  return percentages;
}

export function determineDominant(percentages, dimension) {
  const dimPcts = percentages[dimension];
  return Object.entries(dimPcts).sort((a, b) => b[1] - a[1])[0][0];
}

export function determineSecondary(percentages, dimension, dominant) {
  const dimPcts = percentages[dimension];
  const sorted = Object.entries(dimPcts).sort((a, b) => b[1] - a[1]);
  if (sorted.length < 2) return null;
  const second = sorted[1];
  return second[1] >= dominant[1] * 0.7 ? second[0] : null;
}

export function calculateType(answers, questions) {
  const scores = calculateScores(answers, questions);
  const percentages = calculatePercentages(scores);
  const dims = ['stimulus', 'taste', 'philosophy', 'novelty'];
  const result = {};
  for (const dim of dims) {
    const dominant = determineDominant(percentages, dim);
    const secondary = determineSecondary(percentages, dim, dominant);
    result[dim] = { dominant, secondary, percentages: percentages[dim] };
  }
  return result;
}

// modules/render.js — TasteType 3.1 render engine

export function renderResult(typeCode, typeData, dimensionResults) {
  const typeEl = document.getElementById('result-type-code');
  const nameEl = document.getElementById('result-name');
  const subCodeEl = document.getElementById('result-type-code-sub');
  const selfDescEl = document.getElementById('result-self-description');
  const signatureEl = document.getElementById('result-taste-signature');
  const analysisEl = document.getElementById('result-analysis');
  const avatarEl = document.getElementById('result-avatar');

  if (typeEl) typeEl.textContent = typeCode;
  if (subCodeEl) subCodeEl.textContent = buildSubCode(dimensionResults);
  if (nameEl) nameEl.textContent = typeData.name;
  if (selfDescEl) selfDescEl.textContent = `"${typeData.selfDescription}"`;
  if (analysisEl) analysisEl.textContent = typeData.analysis;

  // Render taste signature tags
  if (signatureEl) {
    signatureEl.innerHTML = '';
    typeData.tasteSignature.forEach(tag => {
      const span = document.createElement('span');
      span.className = 'taste-tag';
      span.textContent = tag;
      signatureEl.appendChild(span);
    });
  }

  // Avatar placeholder
  if (avatarEl) {
    avatarEl.innerHTML = `<span class="avatar-placeholder">[ 头像 ]</span>`;
  }

  // Render details panel
  const detailsPanel = document.getElementById('details-panel');
  if (detailsPanel && dimensionResults) {
    detailsPanel.innerHTML = renderDetailsPanel(dimensionResults);
  }

  // URL state
  const url = new URL(window.location.href);
  url.searchParams.set('type', typeCode);
  window.history.pushState({ type: typeCode }, '', url.toString());
}

function buildSubCode(dimensionResults) {
  const map = { stimulus: {H:'H',N:'n',C:'c',M:'m'}, taste:{U:'U',S:'s',W:'w',B:'b',O:'o'}, philosophy:{A:'A',S:'s'}, novelty:{E:'E',C:'c'} };
  const parts = [];
  for (const dim of ['stimulus', 'taste', 'philosophy', 'novelty']) {
    const r = dimensionResults[dim];
    const code = map[dim][r.dominant];
    const sub = r.secondary ? map[dim][r.secondary] : '';
    parts.push(code + sub);
  }
  return parts.join('-');
}

function renderStackedBars(dimensionResults) {
  const dimLabels = { stimulus:'第一维度 · 刺激偏好', taste:'第二维度 · 核心味觉引力', philosophy:'第三维度 · 风味构建哲学', novelty:'第四维度 · 新奇探索指数' };
  const codeColors = { H:'#E07B54', N:'#C4A4D4', C:'#8BBF8B', M:'#9EB8D0', U:'#E07B54', S:'#C4A4D4', W:'#8BBF8B', B:'#9EB8D0', O:'#F0C87A', A:'#E07B54', E:'#E07B54' };
  const codeLabels = { H:'灼烧', N:'麻感', C:'清凉', M:'温和', U:'鲜味', S:'咸味', W:'甜味', B:'苦醇', O:'酸味', A:'加法烹饪', E:'探索者' };

  let html = '';
  for (const [dim, label] of Object.entries(dimLabels)) {
    const pcts = dimensionResults[dim].percentages;
    const sorted = Object.entries(pcts).sort((a,b) => b[1]-a[1]);
    html += `<div class="dimension-section">
      <div class="dimension-title">${label}</div>
      <div class="stacked-bar-wrap">
        <div class="stacked-bar">${sorted.map(([code,pct])=>`<div class="bar-segment" style="width:${pct}%;background:${codeColors[code]}" title="${code} ${pct}%"></div>`).join('')}</div>
      </div>
      ${sorted.map(([code,pct])=>`<div class="legend-row">
        <div class="legend-left"><div class="legend-dot" style="background:${codeColors[code]}"></div><span class="legend-code">${code}</span><span class="legend-label">(${codeLabels[code]})</span></div>
        <span class="legend-pct">${Math.round(pct)}%</span>
      </div>`).join('')}
    </div>`;
  }
  return html;
}

export function loadTypeSVG(typeCode) {
  return `./assets/${typeCode}.svg`;
